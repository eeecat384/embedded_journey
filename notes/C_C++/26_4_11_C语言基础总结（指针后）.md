# C语言基础总结（指针后）

## 八、指针（Pointer）—— C语言的灵魂

### 8.1 基础概念
```c
int a = 10;              // 普通变量：存储值10，假设地址为0x1000
int *p = &a;             // 指针变量p：存储a的地址0x1000，*表示p是指针

&a  // 取地址运算符：获取变量a的内存地址
*p  // 解引用（间接访问）：通过地址访问指向的内容

printf("%p", &a);        // 打印地址（十六进制，如000000000061FE1C）
printf("%d", *p);        // 打印10（p指向的内容）
*p = 20;                 // 通过指针修改a的值，现在a变成20
```

**指针变量的大小**：与指向的类型无关，64位操作系统为8位，`char*`和`double*`大小相同

### 8.2 指针的类型与声明

`int *`,`double *`,`char *`分别是指向int，double，char的指针><br>
`void *`是万能指针（通用指针），可指向任何类型，但不能直接解引用

```c
// 多重指针（指针的指针）
int **pp;        // 指向int指针的指针（常用于二维数组、动态矩阵）
int ***ppp;      // 三级指针（极少用）

// 注意事项
int *p1, p2;     // p1是指针，p2是int变量！只有p1带*
int *p1, *p2;    // 两者都是指针
```

>局部指针变量 int *p; 未初始化，其值不确定（可能是之前栈帧遗留的任意地址），有安全漏洞<br>
> 初始化时置`NULL`（空指针，0地址，不可访问）<br>
> `int *p = NULL;`

### 8.3 指针与函数
```c
// 使用指针的交换函数（传址调用/引用传递）
void swap(int *pa, int *pb) {
    int temp = *pa;    // 通过解引用获取值
    *pa = *pb;
    *pb = temp;
}

// 通过指针参数带出结果，解决函数只能return一个结果的问题
void divide(int a, int b, int *quotient, int *remainder) {
    *quotient = a / b;    // 通过指针修改外部变量，带出商
    *remainder = a % b;   // 带出余数
}
```

### 8.4 指针与数组

#### 8.4.1 数组名的指针特性
```c
int arr[5] = {10, 20, 30, 40, 50};
int *p = arr;          // 数组名在表达式中退化为指向首元素的指针（&arr[0]）
int len = sizeof(arr); //数组在sizeof中不退化，此处len为20

// 以下等价：
arr[2]     // 数组下标访问，值为30
*(arr + 2) // 指针算术访问，等价于arr[2]
*(p + 2)   // 同上
p[2]       // 指针也可以用下标！完全等价于*(p+2)

```

> 指针的加减：`p + 1`不是地址加1字节，而是加**1个元素大小**（int*则+4字节，double*则+8字节）。编译器根据指针类型自动计算偏移。


#### 8.4.2 指针数组与数组指针
```c
// 指针数组（Array of Pointers）：存储指针的数组
int *parr[5];        // parr是数组，含5个int*元素
// 存储多个字符串（字符指针）
char *names[] = {"Alice", "Bob", "Charlie"};  // 3个char*，指向不同字符串常量

// 数组指针（Pointer to Array）：指向数组的指针
int (*p2arr)[5];     // p2arr是指针，指向含5个int的数组
int arr[5] = {1,2,3,4,5};
p2arr = &arr;        // 必须取整个数组的地址（&arr，不是arr）

(*p2arr)[2] = 30;    // 访问先解引用得到数组，再下标访问，等价于arr[2]=30
```

### 8.5 指针与字符串
```c
// 字符指针和字符数组
char str1[] = "Hello";      // 数组：内容可修改，存储在栈
char *str2 = "Hello";       // 指针：指向字符串常量（只读存储区），不可修改

// 字符串本质就是指针
char s1[20] = "Hello";
char s2[20];
strcpy(s2, s1);             // 复制：从s1地址开始，逐字节复制到s2，直到'\0'
char *s3 = s1;              // s3指向s1，没有复制内容！

// 函数参数中的字符串
void print_str(const char *s) {  // const保护：承诺不修改s指向的内容
    printf("%s", s);
}
```

### 8.6 函数指针
```c
// 定义：指向函数的指针（可存储函数地址，实现回调）
int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }

// 声明函数指针类型
int (*operation)(int, int);  // operation是指向返回int、带两个int参数的函数的指针

operation = add;             // 函数名退化为函数地址
printf("%d", operation(3, 2)); // 调用add(3,2)，输出5

operation = sub;
printf("%d", operation(3, 2)); // 调用sub(3,2)，输出1

```

## 九、结构体、联合体与枚举

### 9.1 结构体定义
结构体就是给程序员自定义数据类型的方法来简化代码
```c
// 定义结构体类型
struct Student {
    int id;
    char name[20];
    float score;
};  

// 定义变量（分配内存）
struct Student s1;                    // 定义
struct Student s2 = {1001, "张三", 85.5};  // 初始化

// 读写成员使用点运算符
s1.id = 1002;
strcpy(s1.name, "李四");
s1.score = 90.0;

// 结构体指针（箭头运算符）
struct Student *ps = &s1;
(*ps).score = 95.0;       // 解引用后用点，麻烦
ps->score = 95.0;         // 推荐使用（->用于通过指针访问成员）

// 结构体数组
struct Student class[50];     // 50个学生
class[0].id = 1;              // 数组第0个元素的id成员
```

#### 9.1.2 结构体嵌套与typedef简化
```c
// typedef：给类型起别名，简化书写
typedef struct Student {
    int id;
    char name[20];
} Stu;  // 现在Stu等价于struct Student

Stu s3;                   // 省写struct
typedef struct Student * StuPtr;  // StuPtr是指向结构体的指针

// 嵌套结构体
struct Date {
    int year, month, day;
};
struct Student {
    int id;
    struct Date birthday;  // 嵌套
};

s1.birthday.year = 2000;  // 逐层访问
```

#### 9.1.3 结构体与指针的重点应用
```c
// 结构体作为函数参数
void print_student2(const Stu *ps) {  
    //传递指针比传递结构体更快，开销也更少，成员多的时候更明显
    printf("%s\n", ps->name);
}

// 结构体与动态内存（结合malloc创建链表节点等，见第11章）
```

> 结构体有内存对齐现象，数据会按照定义时的顺序放到内存地址为数据大小整数倍的地方，且整个结构体大小为最大数据类型的整数倍

### 9.2 联合体（union）—— 共享内存
```c
union Data {
    int i;
    float f;
    char str[20];
};

union Data d;
d.i = 10;           // 写入i
printf("%f", d.f);  // 读取f：i和f共享同一内存，读f得到无意义值

// 所有成员共享起始地址，大小等于最大成员（对齐后）
// 节省内存、类型双关（查看float的二进制表示）、变体类型
```

### 9.3 枚举（enum）—— 命名常量
```c
enum Color { RED, GREEN, BLUE };    // 默认RED=0, GREEN=1, BLUE=2
enum Color c = RED;                 // 声明变量

enum Status { OK = 200, NOT_FOUND = 404, ERROR = 500 };  // 自定义值

// 本质就是int，但增强可读性和类型检查（比#define好）
```

## 十、动态内存管理

### 10.1 核心函数（stdlib.h）
```c
// 分配（Allocate）
int *p = (int *)malloc(5 * sizeof(int));  // 分配5个int的空间（20字节）返回值为void*，需强转为int
// malloc不初始化，内容为垃圾值；失败返回NULL

int *p2 = (int *)calloc(5, sizeof(int));  // 分配并清零（Clear Allocate），适合数组

// 重新分配（Resize）
p = (int *)realloc(p, 10 * sizeof(int));  // 扩缩容，原数据保留，可能移动位置

// 释放（Free）
free(p);         // 归还内存给系统，p变悬空指针（Dangling Pointer）
p = NULL;        // 立即置空，防野指针！
```

### 10.2 使用规范与常见错误
```c
// 标准流程
int *arr = (int *)malloc(5 * sizeof(int));
if (arr == NULL) {           // 内存不足时malloc会失败
    printf("内存分配失败");
    exit(1);
}
// 使用arr...
arr[0] = 10;
free(arr);
arr = NULL;                  // 好习惯，防野指针

//常见错误：
// malloc空间过小，出现越位
// free后出现野指针，安全漏洞
// 内存泄漏：指针生命周期结束后被销毁，分配的内存无法释放
int a = 10;
int *p = &a;
free(p);
// 错误：只能释放malloc/calloc/realloc分配的内存，栈变量不可free
```

### 10.3 动态数组与动态结构体
```c
// 动态数组（运行时决定大小）
int n;
scanf("%d", &n);
int *arr = (int *)malloc(n * sizeof(int));  // C99前变长数组VLA的替代方案

// 动态结构体（链表节点基础）
typedef struct Node {
    int data;
    struct Node *next;
} Node;

Node *head = (Node *)malloc(sizeof(Node));
head->data = 1;
head->next = NULL;
free(head);
```

## 十一、文件操作

### 11.1 文件指针与打开关闭
```c
#include <stdio.h>

FILE *fp;                    // 文件指针（FILE是结构体类型，stdio.h定义）

// 打开文件
fp = fopen("data.txt", "r"); // 文件名，模式
if (fp == NULL) {            // 必须检查（文件不存在、权限不足等）
    perror("打开失败");      // 打印系统错误信息（如"No such file"）
    exit(1);
}

// 模式字符串
// "r"  只读（文件必须存在）
// "w"  只写（文件存在则清空，不存在则创建）
// "a"  追加（写追加到末尾，不清空）
// "r+" 读写（文件必须存在，可随意读写）
// "w+" 读写（创建新文件，或清空已有）
// "a+" 读追加（读可任意位置，写只能追加）
// 加b："rb", "wb" 表示二进制模式（Windows区分文本/二进制，Linux/Mac无区别）

fclose(fp);                  // 关闭文件（必须！缓冲区的内容才会写入磁盘）
// 程序结束时会自动关闭，但显式关闭是好习惯
```

### 11.2 文本文件读写
```c
// 按字符读写
int ch;
while ((ch = fgetc(fp)) != EOF) {  // EOF是文件结束标志，通常-1
    putchar(ch);
}

// 按行读写
char buffer[100];
while (fgets(buffer, 100, fp) != NULL) {  // 读取最多99字符或到换行，含'\n'
    printf("%s", buffer);
}

// 格式化读写（类似scanf/printf）
fprintf(fp, "ID:%d Name:%s Score:%.1f\n", id, name, score);  // 写入
fscanf(fp, "%d %s %f", &id, name, &score);                    // 读取
// 注意：fscanf遇空白停止，不适合含空格的字符串，可用fgets+sscanf组合
```

### 11.3 二进制文件读写（高效，适合结构体）
```c
// 写二进制
Student s = {1001, "张三", 85.5};
fwrite(&s, sizeof(Student), 1, fp);     // 地址，单个大小，个数，文件指针
// 写入s的二进制内存映像（非文本，用记事本打开是乱码）

// 读二进制
Student s2;
fread(&s2, sizeof(Student), 1, fp);     // 从文件读入内存结构

// 批量读写数组
Student class[50];
fwrite(class, sizeof(Student), 50, fp); // 一次写50个学生
```

### 11.4 文件定位与随机访问
```c
fseek(fp, 0, SEEK_SET);    // 移动到开头（SEEK_SET开头，SEEK_CUR当前，SEEK_END末尾）
fseek(fp, -10, SEEK_END);  // 移动到末尾前10字节
long pos = ftell(fp);      // 获取当前位置（相对于开头的偏移字节数）
rewind(fp);                // 回到开头（等价于fseek(fp, 0, SEEK_SET)）

// 示例：读取文件大小
fseek(fp, 0, SEEK_END);
long size = ftell(fp);
rewind(fp);
```

## 十二、其他

### 12.1 位运算（Bitwise Operations）
```c
int a = 5;  // 二进制 00000101
int b = 3;  // 二进制 00000011

int c = a & b;   // 按位与：00000001 (1) - 对应位都为1才为1
int d = a | b;   // 按位或：00000111 (7) - 有1则为1
int e = a ^ b;   // 按位异或：00000110 (6) - 不同则为1（加密常用）
int f = ~a;      // 按位取反：11111010 (-6，补码表示)

// 移位
int g = a << 2;  // 左移2位：00010100 (20)，相当于乘以4
int h = a >> 1;  // 右移1位：00000010 (2)，相当于除以2（整数）
// 有符号数右移：算术右移（补符号位）或逻辑右移（补0），取决于编译器

// 应用：设置/清除/切换特定位
int flags = 0;
flags |= (1 << 3);       // 设置第3位为1
flags &= ~(1 << 3);      // 清除第3位（与第3位为0的掩码相与）
flags ^= (1 << 3);       // 切换第3位（异或）
if (flags & (1 << 3))    // 检查第3位是否为1
```

### 12.2 const与指针的复杂组合
```c
int a = 10, b = 20;

const int *p1 = &a;      // 指向常量的指针：*p1不可改（不能*p1=20），但p1可改指向（p1=&b）
int *const p2 = &a;      // 常量指针：p2不可改（不能p2=&b），但*p2可改（*p2=20）
const int *const p3 = &a;// 指向常量的常量指针：两者都不可改

// 记忆法：const在*左边（*const int），内容(*p)是常量；在*右边（int *const），指针(p)是常量
```
### 12.3内存区域
---
| 区域                    | 存储内容                             | 特点                                     |
| ----------------------- | ------------------------------------ | ---------------------------------------- |
| **代码段**              | 可执行指令（机器码）                 | 只读                                     |
| **数据段**              | 全局变量、静态变量（已初始化）       | 程序加载时分配，生命周期同程序           |
| **BSS段**               | 全局/静态变量（未初始化或初始化为0） | 不占文件空间，运行时清零                 |
| **堆（Heap）**          | malloc分配的动态内存                 | 手动管理，向上增长，空间大               |
| **栈（Stack）**         | 局部变量、函数参数、返回地址         | 自动管理，向下增长，空间有限（通常几MB） |
| **命令行参数/环境变量** | argv, envp                           | 程序启动时传入                           |
